#include<bits/stdc++.h>

using namespace std;

int main(){
    int n;
    cin >> n;
    vector<int> arr(n);
    
    for(int& i : arr)
        cin >> i;

    vector<int> bmax(n);
    bmax[0] = arr[0]+1;
    for(int i=1;i<n;i++)
        bmax[i] = max(bmax[i-1], arr[i]+i+1);

    int emax = 0;
    for(int i=0;i<n;i++){
        cout << max(bmax[n-i-1]+i, emax) << " ";
        emax = 1 + max(arr[n-i-1], emax);
    }
    return 0;
}
